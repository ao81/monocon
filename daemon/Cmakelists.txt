cmake_minimum_required(VERSION 3.16)
project(ArduinoBuildDaemon CXX)

# =============================================================================
# 静的ビルド設定（環境非依存）
# =============================================================================
if(WIN32)
    if(MSVC)
        # MSVCランタイムを静的リンク (/MT)
        set(CMAKE_MSVC_RUNTIME_LIBRARY "MultiThreaded$<$<CONFIG:Debug>:Debug>")
    else()
        # MinGW (GCC) の場合
        add_link_options(-static -static-libgcc -static-libstdc++)
    endif()
endif()

# ----- C++ 標準 -----
set(CMAKE_MSVC_RUNTIME_LIBRARY "MultiThreaded$<$<CONFIG:Debug>:Debug>")
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

# ----- ビルドタイプ -----
if(NOT CMAKE_BUILD_TYPE AND NOT CMAKE_CONFIGURATION_TYPES)
	set(CMAKE_BUILD_TYPE Release CACHE STRING "Build type" FORCE)
endif()

# ----- 最適化フラグ -----
if(MSVC)
	add_compile_options(/W3 /utf-8)
	set(CMAKE_CXX_FLAGS_RELEASE "${CMAKE_CXX_FLAGS_RELEASE} /O2 /Oi /Ot /GL")
	set(CMAKE_EXE_LINKER_FLAGS_RELEASE "${CMAKE_EXE_LINKER_FLAGS_RELEASE} /LTCG")
else()
	add_compile_options(-Wall -Wextra)
	set(CMAKE_CXX_FLAGS_RELEASE "${CMAKE_CXX_FLAGS_RELEASE} -O3")
endif()

# ----- nlohmann/json (FetchContent で自動取得) -----
include(FetchContent)
FetchContent_Declare(
	nlohmann_json
	GIT_REPOSITORY https://github.com/nlohmann/json.git
	GIT_TAG        v3.11.3
)
set(JSON_BuildTests OFF CACHE INTERNAL "")
set(JSON_Install OFF CACHE INTERNAL "")
FetchContent_MakeAvailable(nlohmann_json)

# ----- Windows 固有の設定 -----
if(WIN32)
	# windows.h の min/max マクロを無効化
	# これがないと std::max/std::min が C2589 エラーになる
	add_compile_definitions(NOMINMAX WIN32_LEAN_AND_MEAN)
	# add_compile_definitions(NOMINMAX)
endif()

# =============================================================================
# 共通コード (静的ライブラリ化)
# =============================================================================
add_library(common STATIC
	src/utils.cpp
	src/pipe_io.cpp
	src/port_scanner.cpp
)
target_include_directories(common PUBLIC src)
target_link_libraries(common PUBLIC nlohmann_json::nlohmann_json)
if(WIN32)
	target_link_libraries(common PUBLIC advapi32 setupapi)
endif()

# =============================================================================
# デーモン本体
# =============================================================================
add_executable(arduino-build-daemon
	src/daemon_main.cpp
	src/daemon_state.cpp
	src/builder.cpp
)
target_link_libraries(arduino-build-daemon PRIVATE common)
if(WIN32 AND MSVC)
	set_target_properties(arduino-build-daemon PROPERTIES
		LINK_FLAGS "/SUBSYSTEM:CONSOLE")
elseif(WIN32)
	target_link_options(arduino-build-daemon PRIVATE -mconsole)
endif()

# =============================================================================
# クライアント (薄い CLI)
# =============================================================================
add_executable(arduino-build-cli
	src/client_main.cpp
)
target_link_libraries(arduino-build-cli PRIVATE common)
if(WIN32 AND MSVC)
	set_target_properties(arduino-build-cli PROPERTIES
		LINK_FLAGS "/SUBSYSTEM:CONSOLE")
endif()

# =============================================================================
# 出力先
# =============================================================================
set_target_properties(arduino-build-daemon arduino-build-cli PROPERTIES
	RUNTIME_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/bin"
)
