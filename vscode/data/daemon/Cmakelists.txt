cmake_minimum_required(VERSION 3.21)
# Arduino UNO R4 WiFi (Renesas RA4M1) 向け常駐ビルドデーモン + CLI クライアント
project(arduino-build-tools-unor4wifi CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

# =============================================================================
# ビルドタイプ
# =============================================================================
if(NOT CMAKE_BUILD_TYPE)
    set(CMAKE_BUILD_TYPE Release)
endif()

# =============================================================================
# 最適化フラグ
# =============================================================================
if(MSVC)
    add_compile_options(/O2 /Oi /GL /EHsc /utf-8)
    add_link_options(/LTCG /OPT:REF /OPT:ICF /INCREMENTAL:NO)
else()
    add_compile_options(-O2 -ffunction-sections -fdata-sections)
    add_link_options(-Wl,--gc-sections -s)
endif()

# =============================================================================
# nlohmann/json (FetchContent で自動取得)
# =============================================================================
include(FetchContent)
FetchContent_Declare(
    nlohmann_json
    GIT_REPOSITORY https://github.com/nlohmann/json.git
    GIT_TAG        v3.11.3
)
set(JSON_BuildTests OFF CACHE INTERNAL "")
set(JSON_Install OFF CACHE INTERNAL "")
FetchContent_MakeAvailable(nlohmann_json)

# =============================================================================
# Windows 固有の設定
# =============================================================================
if(WIN32)
    add_compile_definitions(NOMINMAX WIN32_LEAN_AND_MEAN)
endif()

# =============================================================================
# 共通コード (静的ライブラリ化)
# =============================================================================
add_library(common STATIC
    src/utils.cpp
    src/pipe_io.cpp
    src/port_scanner.cpp
    src/ra4m1_uploader.cpp
)
target_include_directories(common PUBLIC src)
target_link_libraries(common PUBLIC nlohmann_json::nlohmann_json)
if(WIN32)
    target_link_libraries(common PUBLIC advapi32 setupapi)
endif()

# =============================================================================
# デーモン本体
# =============================================================================
add_executable(arduino-build-daemon
    src/daemon_main.cpp
    src/daemon_state.cpp
    src/builder.cpp
)
target_link_libraries(arduino-build-daemon PRIVATE common)

if(WIN32 AND MSVC)
    set_target_properties(arduino-build-daemon PROPERTIES
        LINK_FLAGS "/SUBSYSTEM:CONSOLE")
elseif(WIN32)
    target_link_options(arduino-build-daemon PRIVATE -mconsole)
endif()

# =============================================================================
# クライアント (薄い CLI)
# =============================================================================
add_executable(arduino-build-cli
    src/client_main.cpp
)
target_link_libraries(arduino-build-cli PRIVATE common)

if(WIN32 AND MSVC)
    set_target_properties(arduino-build-cli PROPERTIES
        LINK_FLAGS "/SUBSYSTEM:CONSOLE")
endif()

# =============================================================================
# 出力先 (Release/Debug サブフォルダを作らず bin/ 直下に出力)
# =============================================================================
set_target_properties(arduino-build-daemon arduino-build-cli PROPERTIES
    RUNTIME_OUTPUT_DIRECTORY                "${CMAKE_BINARY_DIR}/bin"
    RUNTIME_OUTPUT_DIRECTORY_RELEASE        "${CMAKE_BINARY_DIR}/bin"
    RUNTIME_OUTPUT_DIRECTORY_DEBUG          "${CMAKE_BINARY_DIR}/bin"
    RUNTIME_OUTPUT_DIRECTORY_RELWITHDEBINFO "${CMAKE_BINARY_DIR}/bin"
    RUNTIME_OUTPUT_DIRECTORY_MINSIZEREL     "${CMAKE_BINARY_DIR}/bin"
)

# =============================================================================
# 環境非依存化: 必要なDLLをbin/に同梱
# =============================================================================
if(WIN32)
    set(CMAKE_INSTALL_UCRT_LIBRARIES TRUE)
    set(CMAKE_INSTALL_DEBUG_LIBRARIES FALSE)
    include(InstallRequiredSystemLibraries)

    foreach(target arduino-build-daemon arduino-build-cli)
        if(CMAKE_INSTALL_SYSTEM_RUNTIME_LIBS)
            add_custom_command(TARGET ${target} POST_BUILD
                COMMAND ${CMAKE_COMMAND} -E copy_if_different
                    ${CMAKE_INSTALL_SYSTEM_RUNTIME_LIBS}
                    $<TARGET_FILE_DIR:${target}>
                COMMAND_EXPAND_LISTS
                VERBATIM
            )
        endif()

        add_custom_command(TARGET ${target} POST_BUILD
            COMMAND ${CMAKE_COMMAND} -E copy -t
                $<TARGET_FILE_DIR:${target}>
                $<TARGET_RUNTIME_DLLS:${target}>
            COMMAND_EXPAND_LISTS
            VERBATIM
        )
    endforeach()
endif()

# =============================================================================
# ビルド情報の表示
# =============================================================================
message(STATUS "============================================")
message(STATUS "Build type:    ${CMAKE_BUILD_TYPE}")
message(STATUS "Compiler:      ${CMAKE_CXX_COMPILER_ID}")
message(STATUS "Output dir:    ${CMAKE_BINARY_DIR}/bin")
message(STATUS "============================================")
